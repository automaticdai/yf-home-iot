8.3 Database Request

# execute SQL select statement
cursor.execute("SELECT * FROM LOCATION")

# get the number of rows in the resultset
numrows = int(cursor.rowcount)

# get and display one row at a time.

    row = cursor.fetchone()
    print row[0], "-->", row[1]
